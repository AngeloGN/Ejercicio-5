#include <iostream>
using namespace std;

int main() {
    int a, b, c, d;

    // Calcular las operaciones
    a = -5 + 8 * 6;
    b = (55 + 9) % 9;
    c = 20 + (-3) * 5 / 8;
    d = 5 + 15 / 3 * 2 - 8 % 3;

    // Mostrar los resultados
    cout << "a. -5 + 8 * 6 = " << a << endl;
    cout << "b. (55 + 9) % 9 = " << b << endl;
    cout << "c. 20 + (-3) * 5 / 8 = " << c << endl;
    cout << "d. 5 + 15 / 3 * 2 - 8 % 3 = " << d << endl;

    return 0;
}
